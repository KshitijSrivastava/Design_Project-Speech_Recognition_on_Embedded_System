1function [ H, f, c ] = trifbank( M, K, R, fs, h2w, w2h )
  2% TRIFBANK Triangular filterbank.
  3%
  4%   [H,F,C]=TRIFBANK(M,K,R,FS,H2W,W2H) returns matrix of M triangular filters 
  5%   (one per row), each K coefficients long along with a K coefficient long 
  6%   frequency vector F and M+2 coefficient long cutoff frequency vector C. 
  7%   The triangular filters are between limits given in R (Hz) and are 
  8%   uniformly spaced on a warped scale defined by forward (H2W) and backward 
  9%   (W2H) warping functions.
 10%
 11%   Inputs
 12%           M is the number of filters, i.e., number of rows of H
 13%
 14%           K is the length of frequency response of each filter 
 15%             i.e., number of columns of H
 16%
 17%           R is a two element vector that specifies frequency limits (Hz), 
 18%             i.e., R = [ low_frequency high_frequency ];
 19%
 20%           FS is the sampling frequency (Hz)
 21%
 22%           H2W is a Hertz scale to warped scale function handle
 23%
 24%           W2H is a wared scale to Hertz scale function handle
 25%
 26%   Outputs
 27%           H is a M by K triangular filterbank matrix (one filter per row)
 28%
 29%           F is a frequency vector (Hz) of 1xK dimension
 30%
 31%           C is a vector of filter cutoff frequencies (Hz), 
 32%             note that C(2:end) also represents filter center frequencies,
 33%             and the dimension of C is 1x(M+2)
 34%
 35%   Example
 36%           fs = 16000;               % sampling frequency (Hz)
 37%           nfft = 2^12;              % fft size (number of frequency bins)
 38%           K = nfft/2+1;             % length of each filter
 39%           M = 23;                   % number of filters
 40%
 41%           hz2mel = @(hz)(1127*log(1+hz/700)); % Hertz to mel warping function
 42%           mel2hz = @(mel)(700*exp(mel/1127)-700); % mel to Hertz warping function
 43%
 44%           % Design mel filterbank of M filters each K coefficients long,
 45%           % filters are uniformly spaced on the mel scale between 0 and Fs/2 Hz
 46%           [ H1, freq ] = trifbank( M, K, [0 fs/2], fs, hz2mel, mel2hz );
 47%
 48%           % Design mel filterbank of M filters each K coefficients long,
 49%           % filters are uniformly spaced on the mel scale between 300 and 3750 Hz
 50%           [ H2, freq ] = trifbank( M, K, [300 3750], fs, hz2mel, mel2hz );
 51%
 52%           % Design mel filterbank of 18 filters each K coefficients long, 
 53%           % filters are uniformly spaced on the Hertz scale between 4 and 6 kHz
 54%           [ H3, freq ] = trifbank( 18, K, [4 6]*1E3, fs, @(h)(h), @(h)(h) );
 55%
 56%            hfig = figure('Position', [25 100 800 600], 'PaperPositionMode', ...
 57%                              'auto', 'Visible', 'on', 'color', 'w'); hold on; 
 58%           subplot( 3,1,1 ); 
 59%           plot( freq, H1 );
 60%           xlabel( 'Frequency (Hz)' ); ylabel( 'Weight' ); set( gca, 'box', 'off' ); 
 61%       
 62%           subplot( 3,1,2 );
 63%           plot( freq, H2 );
 64%           xlabel( 'Frequency (Hz)' ); ylabel( 'Weight' ); set( gca, 'box', 'off' ); 
 65%       
 66%           subplot( 3,1,3 ); 
 67%           plot( freq, H3 );
 68%           xlabel( 'Frequency (Hz)' ); ylabel( 'Weight' ); set( gca, 'box', 'off' ); 
 69%
 70%   Reference
 71%           [1] Huang, X., Acero, A., Hon, H., 2001. Spoken Language Processing: 
 72%               A guide to theory, algorithm, and system development. 
 73%               Prentice Hall, Upper Saddle River, NJ, USA (pp. 314-315).
 74
 75%   Author  Kamil Wojcicki, UTD, June 2011
 76
 77
 78%    if( nargin~= 6 ), help trifbank; return; end; % very lite input validation
 79
 80    f_min = 0;          % filter coefficients start at this frequency (Hz)
 81    f_low = R(1);       % lower cutoff frequency (Hz) for the filterbank 
 82    f_high = R(2);      % upper cutoff frequency (Hz) for the filterbank 
 83    f_max = 0.5*fs;     % filter coefficients end at this frequency (Hz)
 84    f = linspace( f_min, f_max, K ); % frequency range (Hz), size 1xK
 85    fw = h2w( f );
 86
 87    % filter cutoff frequencies (Hz) for all filters, size 1x(M+2)
 88    c = w2h( h2w(f_low)+[0:M+1]*((h2w(f_high)-h2w(f_low))/(M+1)) );
 89    cw = h2w( c );
 90
 91    H = zeros( M, K );                  % zero otherwise
 92    for m = 1:M 
 93
 94        % implements Eq. (6.140) on page 314 of [1] 
 95        % k = f>=c(m)&f<=c(m+1); % up-slope
 96        % H(m,k) = 2*(f(k)-c(m)) / ((c(m+2)-c(m))*(c(m+1)-c(m)));
 97        % k = f>=c(m+1)&f<=c(m+2); % down-slope
 98        % H(m,k) = 2*(c(m+2)-f(k)) / ((c(m+2)-c(m))*(c(m+2)-c(m+1)));
 99
100        % implements Eq. (6.141) on page 315 of [1]
101        k = f>=c(m)&f<=c(m+1); % up-slope
102        H(m,k) = (f(k)-c(m))/(c(m+1)-c(m));
103        k = f>=c(m+1)&f<=c(m+2); % down-slope
104        H(m,k) = (c(m+2)-f(k))/(c(m+2)-c(m+1));
105       
106   end
